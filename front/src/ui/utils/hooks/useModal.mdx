# useModal

A utility hook that manages the open/close state of a native HTML `<dialog>` element using the browser's built-in `showModal()` and `close()` APIs.

## Parameters

<table>
  <thead>
    <tr>
      <th>Name</th>
      <th>Type</th>
      <th>Required</th>
      <th>Description</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>
        <code>existingModalRef</code>
      </td>
      <td>
        <code>RefObject&lt;HTMLDialogElement | null&gt;</code>
      </td>
      <td>No</td>
      <td>
        An existing ref pointing to a <code>&lt;dialog&gt;</code> element. If omitted, the hook creates and manages its
        own internal ref.
      </td>
    </tr>
  </tbody>
</table>

      </td>

## Usage

### Basic usage

```tsx
import { useModal } from '@Front/ui/utils/hooks/useModal';

const ConfirmDialog = () => {
  const { modalRef, openModal, closeModal } = useModal();

  return (
    <>
      <button type="button" onClick={openModal}>
        Open dialog
      </button>
      <dialog ref={modalRef}>
        <p>Are you sure?</p>
        <button type="button" onClick={closeModal}>
          Cancel
        </button>
        <button type="button" onClick={closeModal}>
          Confirm
        </button>
      </dialog>
    </>
  );
};
```

### With an existing ref

Use this pattern when the ref needs to be shared with another component or managed externally.

```tsx
import { useRef } from 'react';
import { useModal } from '@Front/ui/utils/hooks/useModal';

const ConfirmDialog = () => {
  const dialogRef = useRef<HTMLDialogElement>(null);
  const { openModal, closeModal } = useModal(dialogRef);

  return (
    <>
      <button type="button" onClick={openModal}>
        Open dialog
      </button>
      <dialog ref={dialogRef}>
        <p>Are you sure?</p>
        <button type="button" onClick={closeModal}>
          Close
        </button>
      </dialog>
    </>
  );
};
```
